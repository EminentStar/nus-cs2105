\begin{cf}{
	\textbf{CSMA/CA} \\
	\vspace{1cm}
	\small{CA = collision avoidance}
}
\end{cf}

\begin{cf}{
	1. No collision detection.\\
	2. Link-layer ACK
}
\end{cf}

\begin{cf}{
	Why no collision detection?\\
	1. $RSSI_{recv}$ < $RSSI_{send}$\\
	2. Cannot detect all collision.\\
}
\end{cf}
